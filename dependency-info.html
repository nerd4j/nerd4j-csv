NERD4J-CSV

So far this is the quickest, easiest and most configurable open source Java framework for CSV manipulation.

Dependency Information

Apache Maven

<dependency>
  <groupId>org.nerd4j</groupId>
  <artifactId>nerd4j-csv</artifactId>
  <version>1.0.1-SNAPSHOT</version>
</dependency>

Apache Buildr

'org.nerd4j:nerd4j-csv:jar:1.0.1-SNAPSHOT'

Apache Ivy

<dependency org="org.nerd4j" name="nerd4j-csv" rev="1.0.1-SNAPSHOT">
  <artifact name="nerd4j-csv" type="jar" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='org.nerd4j', module='nerd4j-csv', version='1.0.1-SNAPSHOT')
)

Grails

compile 'org.nerd4j:nerd4j-csv:1.0.1-SNAPSHOT'

Leiningen

[org.nerd4j/nerd4j-csv "1.0.1-SNAPSHOT"]

SBT

libraryDependencies += "org.nerd4j" % "nerd4j-csv" % "1.0.1-SNAPSHOT"
